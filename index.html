let lift = -10; let velocity = 0; // Pipes let pipes = []; let pipeWidth = 60; let pipeGap = 130; let frame = 0; // Score let score = 0; let gameOver = false; function flap() { if (!gameOver) velocity = lift; else location.reload(); } // Kontrol: Keyboard + Tap layar document.addEventListener("keydown", flap); canvas.addEventListener("touchstart", flap); function drawBird() { ctx.beginPath(); ctx.arc(birdX, birdY, birdRadius, 0, Math.PI * 2); ctx.fillStyle = "yellow"; ctx.fill(); ctx.stroke(); } function drawPipes() { pipes.forEach(pipe => { ctx.fillStyle = "green"; ctx.fillRect(pipe.x, 0, pipeWidth, pipe.top); ctx.fillRect(pipe.x, canvas.height - pipe.bottom, pipeWidth, pipe.bottom); }); } function updatePipes() { if (frame % 90 === 0) { let top = Math.random() * (canvas.height / 2); let bottom = canvas.height - top - pipeGap; pipes.push({ x: canvas.width, top: top, bottom: bottom }); } pipes.forEach(pipe => { pipe.x -= 2; // Score if (pipe.x + pipeWidth === birdX) { score++; } // Collision if ( birdX + birdRadius > pipe.x && birdX - birdRadius < pipe.x + pipeWidth && (birdY - birdRadius < pipe.top || birdY + birdRadius > canvas.height - pipe.bottom) ) { gameOver = true; } }); if (pipes.length && pipes[0].x < -pipeWidth) { pipes.shift(); } } function drawScore() { ctx.fillStyle = "black"; ctx.font = "24px Arial"; ctx.fillText("Score: " + score, 10, 30); if (gameOver) { ctx.fillStyle = "red"; ctx.font = "40px Arial"; ctx.fillText("GAME OVER", 80, canvas.height / 2); ctx.font = "20px Arial"; ctx.fillText("Tap layar / Tekan tombol untuk restart", 30, canvas.height / 2 + 40); } } function update() { ctx.clearRect(0, 0, canvas.width, canvas.height); drawBird(); drawPipes(); drawScore(); velocity += gravity; birdY += velocity; updatePipes(); if (birdY + birdRadius > canvas.height || birdY - birdRadius < 0) { gameOver = true; } frame++; if (!gameOver) requestAnimationFrame(update); else drawScore(); } update();
